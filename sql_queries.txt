#Most common medical conditions
select medical_condition,count(*)as most_common from db.health_care
group by medical_condition
order by most_common;

  #How many unique hospitals are there? How many patients per hospital?
  #Top 10 hospitals with the highest patient count
  select distinct(Hospital)as unique_hsptl,count(*)as patient_count from db.health_care
  group by unique_hsptl
  order by patient_count desc limit 10;
  
  #Most common test results (if quantifiable)?
select test_results,count(*)as most_common from db.health_care
group by test_results 
order by most_common desc;

#What is the total revenue generated by each hospital?
select hospital,sum(billing_amount)as total_revenue from db.health_care
group by hospital
order by total_revenue desc
limit 10;

#Time trends: admissions per month?
SELECT 
  MONTHNAME(STR_TO_DATE(Date_of_Admission, '%Y-%m-%d')) AS month_name,
  COUNT(*) AS monthly_admissions
FROM db.health_care
GROUP BY month_name
ORDER BY FIELD(month_name, 
  'January','February','March','April','May','June',
  'July','August','September','October','November','December');
  
  #Number of patients by gender or city?
select gender,count(*)as number_of_patients from db.health_care
group by gender
order by number_of_patients desc;
